// 1. 备份功能：文件名更专业，方便微信查找 function backupJSON() { const exportData = { brands, data, time: new Date().toLocaleString(), source: "钛云数码对账系统" }; const blob = new Blob([JSON.stringify(exportData, null, 2)], { type: 'application/json' }); const a = document.createElement('a'); const now = new Date(); const dateStr = `${now.getMonth()+1}月${now.getDate()}日_${now.getHours()}点${now.getMinutes()}分`; a.href = URL.createObjectURL(blob); // 文件名示例：钛云数码_报货备份_4月16日_15点30分.json a.download = `钛云数码_报货备份_${dateStr}.json`; a.click(); alert('✅ 备份文件已下载\n您可以通过微信将文件发给同行或保存'); } // 2. 恢复功能：增加微信环境下常见的操作引导 function restoreJSON() { const input = document.createElement('input'); input.type = 'file'; input.accept = '.json'; // 仅限JSON input.onchange = e => { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = ev => { try { const imported = JSON.parse(ev.target.result); // 校验是否是本系统的备份文件 if(imported.brands && imported.data) { brands = imported.brands; data = imported.data; save(); // 保存到本地 init(); // 重新渲染界面 alert('✅ 数据已恢复！所有品牌和单价已同步。'); } else { alert('❌ 恢复失败：该文件不是有效的钛云备份文件'); } } catch(err) { alert('❌ 解析失败：文件损坏或格式不正确'); } }; reader.readAsText(file); }; input.click(); }
